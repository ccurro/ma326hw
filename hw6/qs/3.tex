For each of the following matrices $A \in \mathsf{M}_{n\times n}(F)$,
\begin{enumerate}[(i)]
\item Determine all the eigenvalues of $A$.
\item For each eigenvalue $\lambda$ of $A$, find the set of
  eigenvectors corresponding to $\lambda$.
\item If possible, find a basis for $\mathsf{F}^n$ consisting of
  eigenvectors of $A$.
\item If succesful in finding such a basis, determine and invertible
  matrix $Q$ and a diagonal matrix $D$ such that $Q^{-1}AQ = D$.
\end{enumerate}

\item $\begin{pmatrix}i & 1\\ 2 & -i\end{pmatrix}$ for $F =
  \mathbb{C}$
\begin{enumerate}[(i)]
\item 
\begin{align}
\det{(A-\lambda I)} &= \det{\begin{pmatrix}i-\lambda & 1 \\2 & -i
    -\lambda\end{pmatrix}} =0\\
&= -(i-\lambda)(i+\lambda) -2 = 0\\
&=i^2 -\lambda^2 +2 = 0
\end{align}
\begin{gather}
\lambda^2 = 1\\
\lambda = \pm 1
\end{gather}
\item 
\begin{itemize}
\item For $\lambda=1$
\begin{gather}
Av = \lambda v\\
(A - I)v = 0\\
\begin{pmatrix}

\end{pmatrix}v = 0\\
\begin{pmatrix}

\end{pmatrix}
\leadsto
\begin{pmatrix}

0 & 0
\end{pmatrix}
\end{gather}
\begin{align}
x_1 &= -\frac{t}{i-1}\\
x_2 &= t
\end{align}
\begin{equation}
v = \left\{t\begin{pmatrix}\frac{i+1}{2}\\1\end{pmatrix}\colon t \in \mathbb{C}\right\}
\end{equation}
\item For $\lambda=-1$
