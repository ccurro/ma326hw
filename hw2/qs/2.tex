Determine whether the following sets are linearly dependent or
linearly independent.
\begin{enumerate}
\item[(b)] 
\[
\left\{\begin{pmatrix}
1 & -3\\
-2 & 4
\end{pmatrix}
,
\begin{pmatrix}
-2 & 6\\
4 & -8
\end{pmatrix}\right\} 
\text{in}\;\mathsf{M}_{2\times2}(\mathbb{R})\]

\item[(d)]
\[
\left\{x^3 - , 2x^2 + 4, -2x^3 + 3x^2 + 2x +6\right\} \;\text{in}\;  \mathsf{P}_3\left(\mathbb{R}\right)
\]
\item[(f)]
\[
\left\{\left(1,-1,2\right),\left(1,-2,1\right),\left(1,1,4\right)\right\}
\;\text{in}\; \mathsf{R}^3
\]
\item[(g)]
\[
\left\{
\begin{pmatrix}
1 & 0\\
-2 & 0
\end{pmatrix},
\begin{pmatrix}
0 & -1\\
1 & 1
\end{pmatrix},
\begin{pmatrix}
-1 & 2\\
1 & 0
\end{pmatrix},
\begin{pmatrix}
2 & 1\\
-4 & 4
\end{pmatrix}
\right\}
\;\text{in}\; \mathsf{M}_{2\times2}(\mathbb{R})
\]
\end{enumerate}
